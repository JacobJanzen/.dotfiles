#+title: Neovim Appearance Settings
Use line numbers.
#+begin_src lua :tangle ~/.config/nvim/lua/appearance.lua :mkdirp yes
  vim.opt.number = true
#+end_src

Turn on syntax highlighting.
#+begin_src lua :tangle ~/.config/nvim/lua/appearance.lua :mkdirp yes
  vim.cmd([[
      set termguicolors
      syntax on
      colorscheme default
      set background=light
  ]])
#+end_src
