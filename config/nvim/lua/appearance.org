#+title: Neovim Appearance Settings
Use line numbers.
#+begin_src lua :tangle ~/.config/nvim/lua/appearance.lua :mkdirp yes
  vim.opt.number = true
#+end_src

Set colour scheme to the Ayu light theme.
#+begin_src lua :tangle ~/.config/nvim/lua/appearance.lua :mkdirp yes
  vim.cmd([[
      set termguicolors
      let ayucolor="light"
      syntax on
      colorscheme ayu
  ]])
#+end_src
